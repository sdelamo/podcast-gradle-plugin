18 package groovycalamari.podcast
19 
20 import com.mpatric.mp3agic.Mp3File
21 
22 final class Mp3FileUtils {
23 
24     static Mp3File mp3FileFromUrl(String enclosureUrl) {
25         return new Mp3File(fileFromUrl(enclosureUrl))
26     }
27 
28     private static File fileFromUrl(String urlString) {
29         URL url = new URL(urlString)
30         URLConnection connection = url.openConnection()
31         InputStream inputStream = connection.getInputStream()
32         File file  = File.createTempFile('enclosure', '.mp3')
33         file.deleteOnExit()
34         FileOutputStream fos = new FileOutputStream(file);
35         byte[] buf = new byte[512]
36         while (true) {
37             int len = inputStream.read(buf)
38             if (len == -1) {
39                 break
40             }
41             fos.write(buf, 0, len)
42         }
43         inputStream.close()
44         fos.flush()
45         fos.close()
46         file
47     }
48 }
