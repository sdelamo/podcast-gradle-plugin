217         FileWriter writer = new FileWriter(outputFile)

223     private ItunesPodcast itunesPodcast() {
224         ItunesPodcast.Builder podcastBuilder = iTunesPodcastBuilder()
225 
226         List<MarkdownEpisode> episodes = parseEpisodes(episodes.get())
227         episodes.each {
228             ItunesPodcastEpisode episode = episodeFromMarkdown(it)
229             podcastBuilder.item(episode)
230         }
231 
232         podcastBuilder.build()
233     }

235     private ItunesPodcastEpisode episodeFromMarkdown(MarkdownEpisode markdownEpisode) {
236         ItunesPodcastEpisode.Builder builder = ItunesPodcastEpisode.builder(markdownEpisode.title)
237         if (markdownEpisode.episodeType) {
238             builder.episodeType(markdownEpisode.episodeType)
239         }
240         if (markdownEpisode.guid) {
241             builder.guid(markdownEpisode.guid)
242         }
243         if (markdownEpisode.duration) {
244             builder.duration(markdownEpisode.duration)
245         }
246         if (markdownEpisode.image) {
247             builder.image(markdownEpisode.image)
248         }
249         if (markdownEpisode.episode) {
250             builder.episode(markdownEpisode.episode)
251         }
252         if (markdownEpisode.season) {
253             builder.season(markdownEpisode.season)
254         }
255         if (markdownEpisode.explicit) {
256             builder.explicit(markdownEpisode.explicit)
257         }
258         if (markdownEpisode.author) {
259             builder.author(markdownEpisode.author)
260         }
261         if (markdownEpisode.subtitle) {
262             builder.subtitle(markdownEpisode.subtitle)
263         }
264         if (markdownEpisode.summary) {
265             builder.summary(markdownEpisode.summary)
266         }
267         if (markdownEpisode.content) {
268             builder.contentEncoded(MarkdownUtil.htmlFromMarkdown(markdownEpisode.content))
269         }
270 
271         RssItemEnclosure.Builder enclosureBuilder = RssItemEnclosure.builder()
272         Enclosure enclosure = markdownEpisode.enclosure
273         if (!enclosure.type) {
274             throw new GradleException("Enclosure type for ${markdownEpisode.filename} not present")
275         }
276         if (!enclosure.url) {
277             throw new GradleException("Enclosure url for ${markdownEpisode.filename} not present")
278         }
279         if (!enclosure.length) {
280             throw new GradleException("Enclosure length for ${markdownEpisode.filename} not present")
281         }
282         if (enclosure.url) {
283             enclosureBuilder.url(enclosure.url)
284         }
285         if (enclosure.type) {
286             enclosureBuilder.type(enclosure.type)
287         }
288         if (enclosure.length) {
289             enclosureBuilder.length(enclosure.length)
290         }
291         builder.enclosure(enclosureBuilder.build())
292         builder.build()
293     }

295     private ItunesPodcast.Builder iTunesPodcastBuilder() {
296             ItunesPodcast.Builder podcastBuilder = ItunesPodcast.builder()
297 
298         if (!title.isPresent()) {
299             throw new GradleException("podcast.title is required")
300         }
301         podcastBuilder.title(title.get())
302 
303         if (!link.isPresent()) {
304             throw new GradleException("podcast.link is required")
305         }
306         podcastBuilder.link(link.get())
307 
308         if (!copyright.isPresent()) {
309             throw new GradleException("podcast.copyright is required")
310         }
311         podcastBuilder.copyright(copyright.get())
312 
313         podcastBuilder.language(rssLanguage())
314         if (keywords.isPresent()) {
315             for (String keyword : keywords.get().split(COMMA)) {
316                 podcastBuilder.keyword(keyword)
317             }
318         }
319         if (!about.isPresent()) {
320             throw new GradleException("podcast.description is required")
321         }
322         podcastBuilder.description(about.get())
323 
324         if (subtitle.isPresent()) {
325             podcastBuilder.subtitle(subtitle.get())
326         }
327 
328         if (!author.isPresent()) {
329             throw new GradleException("podcast.author is required")
330         }
331         podcastBuilder.author(author.get())
332 
333         podcastBuilder.type(itunesPodcastType())
334 
335         podcastBuilder.owner(owner())
336         podcastBuilder.image(rssChannelImage())
337         podcastBuilder.block(block.isPresent() ? block.get() : false)
338 
339         if (!explicit.isPresent()) {
340             throw new GradleException("podcast.explicit is required")
341         }
342         podcastBuilder.explict(explicit.get())
343 
344         podcastBuilder.category([(categories.get() as ItunesPodcastCategory).getCategories()])
345 
346         podcastBuilder
347     }

406     List<MarkdownEpisode> parseEpisodes(File folder) {
407         List<MarkdownEpisode> listOfEpisodes = []
408         folder.eachFile { file ->
409             if (file.path.endsWith(SUFFIX_MD) || file.path.endsWith(SUFFIX_MARKDOWN)) {
410                 ContentAndMetadata cm = parseFile(file)
411                 listOfEpisodes << new MarkdownEpisode(filename: file.name, content: cm.content, metadata: cm.metadata)
412             }
413         }
414         listOfEpisodes
415     }

417     static ContentAndMetadata parseFile(File file) {
418         String line = null
419         List<String> lines = []
420         Map<String, String> metadata = [:]
421         boolean metadataProcessed = false
422 
423         int lineCount = 0
424         file.withReader { reader ->
425             while ((line = reader.readLine()) != null) {
426                 if (lineCount == 0 && line.startsWith(SEPARATOR)) {
427                     continue
428                 }
429                 lineCount++
430                 if (!metadataProcessed && line.startsWith(SEPARATOR)) {
431                     metadataProcessed = true
432                     continue
433                 }
434                 if (!metadataProcessed && line.contains(COLON)) {
435                     String metadataKey = line.substring(0, line.indexOf(COLON as String)).trim()
436                     String metadataValue = line.substring(line.indexOf(COLON as String) + COLON.length()).trim()
437                     metadata[metadataKey] = metadataValue
438                 }
439                 if (metadataProcessed) {
440                     lines << line
441                 }
442             }
443         }
444         boolean empty = (!metadataProcessed || lines.isEmpty())
445         new ContentAndMetadata(metadata: empty ? ([:] as Map<String, String>) : metadata,
446                 content: empty ? file.text : lines.join("\n"))
447     }
